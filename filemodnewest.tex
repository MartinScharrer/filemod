\def\test#1{%
  \def\TESTSTR{#1}%
  \edef\TEST{#1}%
  \par\bigskip
  \par\noindent
  \message{^^J%
  \meaning\TESTSTR^^J%
  \meaning\TEST
  ^^J}%
  \meaning\TEST\\
  \TEST
}
